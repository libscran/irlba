31void exact(const Matrix_& matrix, const Eigen::Index requested_number, EigenMatrix_& outU, EigenMatrix_& outV, EigenVector_& outD) {
32 JacobiSVD<EigenMatrix_> svd(matrix.rows(), matrix.cols());

34 auto realizer = matrix.new_known_realize_workspace();

36 svd.compute(realizer->realize(buffer));

38 outD.resize(requested_number);

43

57// Basically (requested * 2 >= smaller), but avoiding overflow from the product.
58inline bool requested_greater_than_or_equal_to_half_smaller(const Eigen::Index requested, const Eigen::Index smaller) {
59 const Eigen::Index half_smaller = smaller / 2;
60 if (requested == half_smaller) {
61 return smaller % 2 == 0;
62 } else {
63 return requested > half_smaller;
64 }

67// Basically min(requested_number + extra_work, smaller), but avoiding overflow from the sum.
68inline Eigen::Index choose_requested_plus_extra_work_or_smaller(const Eigen::Index requested_number, const Eigen::Index extra_work, const Eigen::Index smaller) {
69 if (requested_number >= smaller) {
70 return smaller;
71 } else {
72 // This is guaranteed to fit into an Eigen::Index;
73 // either it's equal to 'smaller' or it's less than 'smaller'.
74 return requested_number + sanisizer::min(smaller - requested_number, extra_work);
75 }
76}

78// Setting 'k'. This looks kinda weird, but this is deliberate, see the text below Algorithm 3.1 of Baglama and Reichel.
79//
80// - Our n_converged is their k'.
81// - Our requested_number is their k.
82// - Our work is their m.
83// - Our returned k is their k + k''.
84//
85// They don't mention anything about the value corresponding to our input k, but I guess we always want k to increase.
86// So, if our proposed new value of k is lower than our current value of k, we just pick the latter.
87inline Eigen::Index update_k(Eigen::Index k, const Eigen::Index requested_number, const Eigen::Index n_converged, const Eigen::Index work) {
88 // Here, the goal is to reproduce this code from irlba's convtests() function, but without any risk of wraparound or overflow.
89 //
90 // if (k < requested_number + n_converged) {
91 // k = requested_number + n_converged;
92 // }
93 // if (k > work - 3) {
94 // k = work - 3;
95 // }
96 // if (k < 1) {
97 // k = 1;
98 // }
99
100 if (work <= 3) {
101 return 1;
102 }
103 const Eigen::Index limit = work - 3;
104
105 const auto less_than_requested_plus_converged = [&](const Eigen::Index val) -> bool {
106 return val < requested_number || static_cast<Eigen::Index>(val - requested_number) < n_converged;
107 };
108
109 if (less_than_requested_plus_converged(k)) {
110 if (less_than_requested_plus_converged(limit)) {
111 return limit;
112 } else {
113 const Eigen::Index output = requested_number + n_converged;
114 return std::max(output, static_cast<Eigen::Index>(1));
115 }
116 } else {
117 const Eigen::Index output = std::min(k, limit);
118 return std::max(output, static_cast<Eigen::Index>(1));
119 }

170template<class Matrix_, class EigenMatrix_, class EigenVector_>
171Metrics compute(
172 const Matrix_& matrix,
173 const Eigen::Index number,
174 EigenMatrix_& outU,
175 EigenMatrix_& outV,
176 EigenVector_& outD,
177 const Options<EigenVector_>& options
178) {
179 const Eigen::Index smaller = std::min(matrix.rows(), matrix.cols());
180 Eigen::Index requested_number = sanisizer::cast<Eigen::Index>(number);
181 if (requested_number > smaller) {
182 if (options.cap_number) {
183 requested_number = smaller;
184 } else {
185 throw std::runtime_error("requested number of singular values cannot be greater than the smaller matrix dimension");
186 }
187 } else if (requested_number == smaller && !options.exact_for_large_number) {
188 throw std::runtime_error("requested number of singular values must be less than the smaller matrix dimension for IRLBA iterations");
189 }
190
191 // Optimization if no SVs are requested. Also protect the exact SVD code
192 // from a zero-extent matrix, as this causes Eigen to shit itself.
193 if (requested_number == 0) {
194 outD.resize(requested_number);
195 outU.resize(matrix.rows(), requested_number);
196 outV.resize(matrix.cols(), requested_number);
197 Metrics met;
198 met.converged = true;
199 return met;
200 }
201
202 // Falling back to an exact SVD for small matrices or if the requested number is too large
203 // (not enough of a workspace). Hey, I don't make the rules.
204 if (
205 (options.exact_for_small_matrix && smaller < 6) ||
206 (options.exact_for_large_number && requested_greater_than_or_equal_to_half_smaller(requested_number, smaller))
207 ) {
208 exact(matrix, requested_number, outU, outV, outD);
209 Metrics met;
210 met.converged = true;
211 return met;
212 }
213
214 const Eigen::Index extra_work = choose_extra_work(requested_number, options.extra_work);
215
216 // We know work must be positive at this point, as we would have returned early if requested_number = 0.
217 // Similar, if smaller = 0, we would have either thrown earlier or set requested_number = 0.
218 const Eigen::Index work = choose_requested_plus_extra_work_or_smaller(requested_number, extra_work, smaller);
219
220 // Don't worry about sanitizing dimensions for Eigen constructors,
221 // as the former are stored as Eigen::Index and the latter accepts Eigen::Index inputs.
222 EigenMatrix_ V(matrix.cols(), work);
223 std::mt19937_64 eng(options.seed);
224 if (options.initial.has_value()) {
225 const auto& init = *(options.initial);
226 if (init.size() != V.rows()) {
227 throw std::runtime_error("initialization vector does not have expected number of rows");
228 }
229 V.col(0) = init;
230 } else {
231 fill_with_random_normals(V, 0, eng);
232 }
233 V.col(0) /= V.col(0).norm();
234
235 bool converged = false;
236 int iter = 0, mult = 0;
237 Eigen::Index k = 0;
238 JacobiSVD<EigenMatrix_> svd(work, work);
239
240 LanczosWorkspace<EigenVector_, Matrix_> lpwork(matrix);
241
242 EigenMatrix_ W(matrix.rows(), work);
243 EigenMatrix_ Wtmp(matrix.rows(), work);
244 EigenMatrix_ Vtmp(matrix.cols(), work);
245
246 EigenMatrix_ B(work, work);
247 B.setZero(work, work);
248 EigenVector_ res(work);
249 EigenVector_ F(matrix.cols());
250
251 EigenVector_ prevS(work);
252 typename EigenMatrix_::Scalar svtol = options.singular_value_ratio_tolerance;
253 typename EigenMatrix_::Scalar tol = options.convergence_tolerance;
254 typename EigenMatrix_::Scalar svtol_actual = (svtol >= 0 ? svtol : tol);
255
256 for (; iter < options.max_iterations; ++iter) {
257 // Technically, this is only a 'true' Lanczos bidiagonalization
258 // when k = 0. All other times, we're just recycling the machinery,
259 // see the text below Equation 3.11 in Baglama and Reichel.
260 mult += run_lanczos_bidiagonalization(lpwork, W, V, B, eng, k, options);
261
262// if (iter < 2) {
263// std::cout << "B is currently:\n" << B << std::endl;
264// std::cout << "W is currently:\n" << W << std::endl;
265// std::cout << "V is currently:\n" << V << std::endl;
266// }
267
268 svd.compute(B);
269 const auto& BS = svd.singularValues();
270 const auto& BU = svd.matrixU();
271 const auto& BV = svd.matrixV();
272
273 // Checking for convergence.
274 if (B(work - 1, work - 1) == 0) { // a.k.a. the final value of 'S' from the Lanczos iterations.
275 converged = true;
276 break;
277 }
278
279 const auto& F = lpwork.F; // i.e., the residuals, see Algorithm 2.1 of Baglama and Reichel.
280 auto R_F = F.norm();
281
282 // Computes the convergence criterion defined in on the LHS of
283 // Equation 2.13 of Baglama and Riechel. (e_m is literally the unit
284 // vector along the m-th dimension, so it ends up just being the
285 // m-th row of the U matrix.) We expose it here as we will be
286 // re-using the same values to update B, see below.
287 res = R_F * BU.row(work - 1);
288
289 Eigen::Index n_converged = 0;
290 if (iter > 0) {
291 const auto Smax = *std::max_element(BS.begin(), BS.end());
292 const auto threshold = Smax * tol;
293
294 for (Eigen::Index j = 0; j < work; ++j) {
295 if (std::abs(res[j]) <= threshold) {
296 const auto ratio = std::abs(BS[j] - prevS[j]) / BS[j];
297 if (ratio <= svtol_actual) {
298 ++n_converged;
299 }
300 }
301 }
302
303 if (n_converged >= requested_number) {
304 converged = true;
305 break;
306 }
307 }
308 prevS = BS;
309
310 k = update_k(k, requested_number, n_converged, work);
311 Vtmp.leftCols(k).noalias() = V * BV.leftCols(k);
312 V.leftCols(k) = Vtmp.leftCols(k);
313
314 // See Equation 3.2 of Baglama and Reichel, where our 'V' is their
315 // 'P', and our 'F / R_F' is their 'p_{m+1}' (Equation 2.2). 'F'
316 // was orthogonal to the old 'V' and so it should still be
317 // orthogonal to the new left-most columns of 'V'; the input
318 // expectations of the Lanczos bidiagonalization are still met, so
319 // V is ok to use in the next run_lanczos_bidiagonalization().
320 V.col(k) = F / R_F;
321
322 Wtmp.leftCols(k).noalias() = W * BU.leftCols(k);
323 W.leftCols(k) = Wtmp.leftCols(k);
324
325 B.setZero(work, work);
326 for (I<decltype(k)> l = 0; l < k; ++l) {
327 B(l, l) = BS[l];
328
329 // This assignment looks weird but is deliberate, see Equation
330 // 3.6 of Baglama and Reichel. Happily, this is the same value
331 // used to determine convergence in 2.13, so we just re-use it.
332 // (See the equation just above Equation 3.5; I think they
333 // misplaced a tilde on the final 'u', given no other 'u' has
334 // a B_m superscript as well as a tilde.)
335 B(l, k) = res[l];
336 }
337 }
338
339 // See Equation 2.11 of Baglama and Reichel for how to get from B's
340 // singular triplets to mat's singular triplets.
341 outD.resize(requested_number);
342 outD = svd.singularValues().head(requested_number);
343
344 outU.resize(matrix.rows(), requested_number);
345 outU.noalias() = W * svd.matrixU().leftCols(requested_number);
346
347 outV.resize(matrix.cols(), requested_number);
348 outV.noalias() = V * svd.matrixV().leftCols(requested_number);
349
350 Metrics met;
351 met.converged = converged;
352 met.multiplications = mult;
353 met.iterations = (converged ? iter + 1 : iter);
354 return met;
355}
